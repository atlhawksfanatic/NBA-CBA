  entryformat=\bfseries,
  pagenumberformat=\bfseries,
]{tocline}{chapter}

  dynnumwidth
]{tocline}{%
  chapter,section,subsection,subsubsection,paragraph,subparagraph,%
  figure,table
}
\DeclareTOCStyleEntries[
  dynindent
]{tocline}{subsection,subsubsection,subparagraph}
